Pregunta 1


table3 %>%
    separate(__________, 
             c("cases", "population"), 
             sep="__")

Solución 1


table3 %>%
    separate(rate, #columna para dividir los datos
             c("cases", "population"), #nuevas columnas
             sep="/") #separador

Pregunta 2


table1 %>%
  unite("rate",
        _____, _________,
        sep="__")

Solución 2


table1 %>%
  unite("rate",
        cases, population,
        sep="/")
